#include <stdio.h>
int piramide(){
int n,fila,col;
printf("tamaño de la piramide: ");
scanf("%d",&n);

for(fila=1;fila<=n;fila++){
/* Espacios en blanco*/
for(col=1;col<=n-fila;col++){
printf(" ");
}
for(col=1;col<=2*fila-1;col++){
printf("*");
}
printf("\n");
}
}

int numeromayor(){
int i,mayor=0;

do{
printf("\nNúmero: ");
scanf("%i",&i);
if (mayor<i) mayor=i;
}while (i!=0);

printf("\nEl mayor es %i",mayor);
}

void main(){
	piramide();
	numeromayor();
}
